  13 // Decodes YOLO output tensor (1 x N x (4+K) or 1 x (4+K) x N) into per-detection arrays on device.
  14 // Inputs:
  15 //   d_out: device pointer to model output float tensor (assumed contiguous)
  16 //   num_det, num_attrs, num_classes: tensor layout params
  17 //   channels_first: if true, layout is [C,N], else [N,C]
  18 //   conf_thr: score threshold
  19 //   letterbox meta: scale/pad and original size for mapping back
  20 // Outputs:
  21 //   d_boxes: device float array [M,4] (yxyx) (pre-allocated for num_det)
  22 //   d_scores: device float array [M]
  23 //   d_classes: device int32 array [M]
  24 //   d_count: device int, number of written detections (<= num_det)
  25 // Returns cudaError_t
  26 cudaError_t yolo_decode_to_yxyx(
  27     const float* d_out,
  28     int num_det,
  29     int num_attrs,
  30     int num_classes,
  31     int channels_first,
  32     float conf_thr,
  33     float scale,
  34     int pad_x,
  35     int pad_y,
  36     int orig_w,
  37     int orig_h,
  38     float* d_boxes,
  39     float* d_scores,
  40     int32_t* d_classes,
  41     int* d_count,
  42     cudaStream_t stream);
  43
